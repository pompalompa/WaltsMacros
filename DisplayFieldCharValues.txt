' 64-bit Windows 10 Pro, but no guarantees are promised or implied.
'
'****************************************************************************************
' How it works: Run the macro to see the numeric values (OCLC-MARC, not ASCII) of the
' characters in the field, displayed in pages containing up to 72 characters. Characters
' not able to be displayed by the system font are shown with the "not sign" substituted,
' except that the delimiter sign is represented by "[$]" and super- and subscript
' characters are shown surrounded by middle dots.
'****************************************************************************************

Option Explicit

Global CurrentPage%

Global Chars() As String

Tag$          = Left$( FieldData$, 3 )

ReDim Chars( 1 )
ReDim Codes( 1 )

' Go through the string of data from the field, testing each 8-byte chunk to see if it's
' Unicode. If it is, strip the formatting characters from beginning and end, keeping only
' the four-byte code. Assign it the ASCII value of zero to make it display as "[]". If
' the 8 bytes don't indicate Unicode, take just the first byte and treat it as ASCII.

Start% = p
Do
  UnicodeTest$ = Mid$( FieldData$, p, 8 )
  ReDim Preserve Codes( i )
  ReDim Preserve Chars( i )
  If Left$( UnicodeTest$, 3 ) = "&#x" and Right$( UnicodeTest$, 1 ) = ";" Then
      Codes( i ) = Mid$( UnicodeTest$, 4, 4 )

      p = p + 8

      p = p + 1

' those that display in a dialog box. Unicode characters will all display as "[]".

    Case 32 To 37, 39 To 126         'Plain low ASCII

    Case 38
      Chars( i ) = "&&"
    Case 128 To 137        'Superscript numbers
      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 80 ) & Chr$( 183 )
    Case 138 To 139        'Superscript parentheses
      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 98 ) & Chr$( 183 )
    Case 140               'Superscript plus
      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 97 ) & Chr$( 183 )
    Case 141, 144 To 153   'Superscript minus, subscript numbers
      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 96 ) & Chr$( 183 )
    Case 154 To 155        'Subscript parentheses
      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 114 ) & Chr$( 183 )
    Case 156               'Subscript plus
      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 111 ) & Chr$( 183 )

      Chars( i ) = Chr$( 183 ) & Chr$( ASCIIChar% - 114 ) & Chr$( 183 )
    Case 158               'Degree sign
      Chars( i ) = Chr$( 176 )
    Case 159               'sharp S
      Chars( i ) = Chr$( 223 )
    Case 160               'Euro
      Chars( i ) = Chr$( 128 )
    Case 162               'Crossed O
      Chars( i ) = Chr$( 216 )
    Case 163               'Eth
      Chars( i ) = Chr$( 208 )
    Case 164               'Thorn
      Chars( i ) = Chr$( 222 )
    Case 165               'AE Ligature
      Chars( i ) = Chr$( 198 )
    Case 166               'OE Ligature
      Chars( i ) = Chr$( 140 )
    Case 168               'Middle dot
      Chars( i ) = Chr$( 183 )
    Case 170               'registered
      Chars( i ) = Chr$( 174 )
    Case 171               'Plus-minus
      Chars( i ) = Chr$( 177 )
    Case 178               'crossed o
      Chars( i ) = Chr$( 248 )
    Case 179               'eth
      Chars( i ) = Chr$( 240 )
    Case 180               'thorn
      Chars( i ) = Chr$( 254 )
    Case 181               'ae ligature
      Chars( i ) = Chr$( 230 )
    Case 182               'oe ligature
      Chars( i ) = Chr$( 156 )
    Case 185               'pound
      Chars( i ) = Chr$( 163 )
    Case 186               'eth
      Chars( i ) = Chr$( 240 )
    Case 200               'inverted question mark
      chars( i ) = Chr$( 191 )
    Case 201               'inverted exclamation mark
      chars( i ) = Chr$( 161 )
    Case 202               'copyright
      Chars( i ) = Chr$( 169 )
    Case 223               'delimiter
      Chars( i ) = "[$]"
    Case 226               'acute
      Chars( i ) = Chr$( 180 )
    Case 227               'circumflex
      Chars( i ) = Chr$( 136 )
    Case 228               'tilde
      Chars( i ) = Chr$( 152 )
    Case 229               'macron
      Chars( i ) = Chr$( 175 )
    Case 232               'Diaresis
      Chars( i ) = Chr$( 168 )
    Case 234               'circle above letter
      Chars( i ) = Chr$( 176 )
    Case 240               'cedilla
      Chars( i ) = Chr$( 184 )

' Characters with no equivalent or substitute get an empty pair of brackets (that looks
' somewhat like a box) to avoid a variety of junk characters appearing on the screen.

      Chars( i ) = "[]"

  i = i + 1

'****************************************************************************************

Function Dialog1ControlDefinition( Id$, Action%, SVal& )

Const CONTROL_CHANGE As Integer = 2   'The value of the dialog box function parameter "Action%" when a control changes.
Const INITIALIZE     As Integer = 1   'The value of the dialog box function parameter "Action%" when the dialog box opens.
Const KEEP_DLG_OPEN  As Integer = -1  'The value of the dialog box function to keep a dialog box open.

Dim FirstVal%
Dim LastVal%
Dim Range$

Select Case Action%

  Case INITIALIZE

DifferentPage:    'When a new page is to be displayed, the dialog box cycles back to this line.

' Control display of the "Previous" and "Next" buttons: at the ends of the ranges, they
' need to be appropriately disabled.

    If CurrentPage% = 1 Then
        If TotalPages% = 1 Then
            DlgVisible "Prev",   0
            DlgVisible "Next",   0
          Else
            DlgEnable  "Prev",   0
            DlgEnable  "Next",   1
        End If
      Else
        If CurrentPage% = TotalPages% Then
            DlgEnable  "Prev",   1
            DlgEnable  "Next",   0
          Else
            DlgEnable  "Prev",   1
            DlgEnable  "Next",   1
        End If
    End If

    DlgValue   "Pages",  CurrentPage% - 1

' Hide the row boxes for empty rows on the last page.

    If CurrentPage% = TotalPages% Then
        If Remainder% < 13 Then
            DlgVisible "Row2",   0
            DlgVisible "Row3",   0
            DlgVisible "Row4",   0
            DlgVisible "Row5",   0
            DlgVisible "Row6",   0
            DlgVisible "Row7",   0
            DlgVisible "Row8",   0
          ElseIf Remainder% < 25 Then
            DlgVisible "Row3",   0
            DlgVisible "Row4",   0
            DlgVisible "Row5",   0
            DlgVisible "Row6",   0
            DlgVisible "Row7",   0
            DlgVisible "Row8",   0
          ElseIf Remainder% < 37 Then
            DlgVisible "Row4",   0
            DlgVisible "Row5",   0
            DlgVisible "Row6",   0
            DlgVisible "Row7",   0
            DlgVisible "Row8",   0
          ElseIf Remainder% < 49 Then
            DlgVisible "Row5",   0
            DlgVisible "Row6",   0
            DlgVisible "Row7",   0
            DlgVisible "Row8",   0
          ElseIf Remainder% < 61 Then
            DlgVisible "Row6",   0
            DlgVisible "Row7",   0
            DlgVisible "Row8",   0
          ElseIf Remainder% < 73 Then
            DlgVisible "Row7",   0
            DlgVisible "Row8",   0
          ElseIf Remainder% < 85 Then
            DlgVisible "Row8",   0
        End If
      Else
        DlgVisible "Row1",   1
        DlgVisible "Row2",   1
        DlgVisible "Row3",   1
        DlgVisible "Row4",   1
        DlgVisible "Row5",   1
        DlgVisible "Row6",   1
        DlgVisible "Row7",   1
        DlgVisible "Row8",   1
    End If

    DlgText    "Char1",  Chars(  1 + ( PAGE_LIMIT * ( CurrentPage% - 1 ) ) )
    DlgText    "Code1",  Codes(  1 + ( PAGE_LIMIT * ( CurrentPage% - 1 ) ) )
    DlgText    "Char2",  Chars(  2 + ( PAGE_LIMIT * ( CurrentPage% - 1 ) ) )
